CREATE TABLE `searchservice`.`searchdata` (
  `KEYNUM` INT NOT NULL AUTO_INCREMENT,
  `SEARCHKEY` VARCHAR(64) NOT NULL,
  `CATEID` VARCHAR(4) NOT NULL,
  `ITEMID` VARCHAR(8) NOT NULL,
  `SEARCHCOUNT` INT NOT NULL DEFAULT 0,
  `CLICKCOUNT` INT NOT NULL DEFAULT 0,
  `CREATEDTIME` DATETIME NOT NULL,
  `MODIFIEDTIME` DATETIME NOT NULL,
  PRIMARY KEY (`KEYNUM`),
  UNIQUE INDEX `SEARCHKEY_UNIQUE` (`SEARCHKEY` ASC) VISIBLE);
